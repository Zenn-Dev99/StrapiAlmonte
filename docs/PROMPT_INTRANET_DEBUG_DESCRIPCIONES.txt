function textoAHTML(texto) {
  if (!texto || texto.trim() === '') return '';
  
  // Si ya tiene HTML, devolverlo
  if (texto.includes('<p>') || texto.includes('<div>')) {
    return texto;
  }
  
  // Convertir texto plano a HTML
  const parrafos = texto.split('\n\n')
    .filter(p => p.trim())
    .map(p => `<p>${p.trim().replace(/\n/g, '<br>')}</p>`)
    .join('');
  
  return parrafos || `<p>${texto}</p>`;
}

  let textoLimpio = texto.replace(/<[^>]+>/g, '').trim();
  textoLimpio = textoLimpio.replace(/\s+/g, ' ');
  
  if (textoLimpio.length <= maxCaracteres) {
    return `<p>${textoLimpio}</p>`;
  }
  
  let textoCorto = textoLimpio.substring(0, maxCaracteres);
  const ultimoEspacio = textoCorto.lastIndexOf(' ');
  
  if (ultimoEspacio > maxCaracteres * 0.8) {
    textoCorto = textoCorto.substring(0, ultimoEspacio);
  }
  
  return `<p>${textoCorto.trim()}...</p>`;
}
